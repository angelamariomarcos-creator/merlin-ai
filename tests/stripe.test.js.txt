const request = require('supertest');
const app = require('../server'); // Enlace al servidor central de Express

describe('Stripe Integration', () => {
    
    test('POST /checkout/create-session requires promptId', async () => {
        const res = await request(app)
            .post('/checkout/create-session')
            .send({ userId: 'test@test.com' });
        
        expect(res.statusCode).toBe(404);
    });
 
    test('POST /checkout/create-session returns sessionId', async () => {
        const res = await request(app)
            .post('/checkout/create-session')
            .send({
                promptId: 1,
                userId: 'test@test.com'
            });
        
        expect(res.statusCode).toBe(200);
        expect(res.body.sessionId).toBeDefined();
    });
});
